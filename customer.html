function checkConnection() { const url = SCRIPT_URL + '?action=getDashboardStats'; fetch(url) .then(response => response.text()) .then(text => { document.getElementById('debugOutput').textContent = text; try { const data = JSON.parse(text); if (data.stats) { updateStatus(true, 'Connected'); } else { updateStatus(false, 'Invalid response'); } } catch (e) { updateStatus(false, 'Not JSON'); } }) .catch(error => { document.getElementById('debugOutput').textContent = 'Fetch error: ' + error; updateStatus(false, 'Fetch failed'); }); }
