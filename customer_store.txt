    <script>
        // ========== DATA ==========
        let products = [];
        let cart = JSON.parse(localStorage.getItem('sol_cart')) || [];
        let currentFilter = 'all';
        let searchQuery = '';
        let selectedPayment = 'cod';

        // Load products from localStorage (synced with admin panel)
        function loadProducts() {

                    { id: 'SOL009', name: 'Bridal Bangles Set', category: 'Bangles', price: 3600, stock: 'In Stock', img: 'https://images.unsplash.com/photo-1611591437281-460bfbe1220a?ixlib=rb-1.2.1&auto=format&fit=crop&w=300&q=80' },

            }
            renderProducts();

        // Render products
        function renderProducts() {
            const grid = document.getElementById('productsGrid');
            
            let filtered = [...products];
            
            // Apply category filter
            if (currentFilter !== 'all') {
                filtered = filtered.filter(p => p.category === currentFilter);
            }
            
            // Apply search filter
            if (searchQuery) {
                filtered = filtered.filter(p => 
                    p.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
                    p.id.toLowerCase().includes(searchQuery.toLowerCase())
                );
            }

            if (filtered.length === 0) {
                grid.innerHTML = `
                    <div class="empty-state" style="grid-column: 1/-1;">
                        <i class="fas fa-gem"></i>
                        <h3>No products found</h3>
                        <p>Try adjusting your filters or search query</p>
                    </div>
                `;
                return;
            }

            let html = '';
            filtered.forEach((p, index) => {
                let stockClass = 'stock-instock';
                if (p.stock === 'Limited') stockClass = 'stock-limited';
                if (p.stock === 'Out of Stock') stockClass = 'stock-out';

                const imgHtml = p.img ? 
                    `<img class="product-img" src="${p.img}" alt="${p.name}" onerror="this.src='https://via.placeholder.com/300x300/D4AF37/ffffff?text=SOL'">` : 
                    `<div class="product-img-placeholder"><i class="fas fa-gem"></i></div>`;

                html += `
                    <div class="product-card" data-product-id="${p.id}">
                        <div class="product-badge">${p.category}</div>
                        ${imgHtml}
                        <div class="product-info">
                            <div class="product-id">${p.id}</div>
                            <div class="product-name">${p.name}</div>
                            <div class="product-price">${p.price}</div>
                            <div class="product-stock ${stockClass}">${p.stock}</div>
                            <div class="product-actions">
                                <button class="add-to-cart-btn" onclick="addToCart('${p.id}')" ${p.stock === 'Out of Stock' ? 'disabled' : ''}>
                                    <i class="fas fa-shopping-cart"></i> Add
                                </button>
                                <button class="buy-now-btn" onclick="buyNow('${p.id}')" ${p.stock === 'Out of Stock' ? 'disabled' : ''}>
                                    <i class="fas fa-bolt"></i> Buy
                                </button>
                            </div>
                        </div>
                    </div>
                `;
            });
            grid.innerHTML = html;
        }

        // Filter products
        window.filterProducts = function(category) {
            currentFilter = category;
            
            // Update active filter button
            document.querySelectorAll('.filter-btn').forEach(btn => {
                btn.classList.remove('active');
                if (btn.textContent.toLowerCase() === category.toLowerCase() || 
                    (category === 'all' && btn.textContent === 'All')) {
                    btn.classList.add('active');
                }
            });
            
            renderProducts();
        };

        // Search products
        window.searchProducts = function() {
            searchQuery = document.getElementById('searchInput').value;
            renderProducts();
        };

        // Scroll to products
        window.scrollToProducts = function() {
            document.getElementById('products').scrollIntoView({ behavior: 'smooth' });
        };

        // Add to cart
        window.addToCart = function(productId) {
            const product = products.find(p => p.id === productId);
            if (!product || product.stock === 'Out of Stock') return;

            const existingItem = cart.find(item => item.id === productId);
            
            if (existingItem) {
                existingItem.quantity += 1;
            } else {
                cart.push({
                    id: product.id,
                    name: product.name,
                    price: product.price,
                    img: product.img,
                    quantity: 1
                });
            }

            // Save to localStorage
            localStorage.setItem('sol_cart', JSON.stringify(cart));
            
            // Update UI
            updateCartCount();
            renderCart();
            showNotification(`✨ ${product.name} added to cart!`);
            
            // Animate cart icon
            const cartIcon = document.querySelector('.cart-icon');
            cartIcon.style.transform = 'scale(1.2)';
            setTimeout(() => {
                cartIcon.style.transform = 'scale(1)';
            }, 200);
        };

        // Buy now (add to cart and open checkout)
        window.buyNow = function(productId) {
            addToCart(productId);
            setTimeout(() => {
                toggleCart();
                openCheckout();
            }, 500);
        };

        // Render cart items
        function renderCart() {
            const cartDiv = document.getElementById('cartItems');
            const totalSpan = document.getElementById('cartTotal');
            
            if (cart.length === 0) {
                cartDiv.innerHTML = `
                    <div class="empty-state">
                        <i class="fas fa-shopping-bag"></i>
                        <p>Your cart is empty</p>
                        <p style="font-size: 0.9rem; margin-top: 10px;">Add some beautiful jewelry to get started!</p>
                    </div>
                `;
                totalSpan.textContent = '₹0';
                return;
            }

            let html = '';
            let total = 0;

            cart.forEach((item, index) => {
                total += item.price * item.quantity;
                
                const imgHtml = item.img ? 
                    `<img class="cart-item-img" src="${item.img}" alt="${item.name}">` : 
                    `<div class="cart-item-img" style="background: #f0e8e0; display: flex; align-items: center; justify-content: center;"><i class="fas fa-gem" style="color: var(--primary-gold);"></i></div>`;

                html += `
                    <div class="cart-item">
                        ${imgHtml}
                        <div class="cart-item-details">
                            <div class="cart-item-name">${item.name}</div>
                            <div class="cart-item-price">₹${item.price} × ${item.quantity}</div>
                            <div class="cart-controls">
                                <button onclick="updateQuantity(${index}, -1)">−</button>
                                <span>${item.quantity}</span>
                                <button onclick="updateQuantity(${index}, 1)">+</button>
                                <button class="remove-item" onclick="removeItem(${index})">
                                    <i class="fas fa-trash"></i>
                                </button>
                            </div>
                        </div>
                    </div>
                `;
            });

            cartDiv.innerHTML = html;
            totalSpan.textContent = `₹${total}`;
        }

        // Update quantity
        window.updateQuantity = function(index, change) {
            if (cart[index]) {
                cart[index].quantity += change;
                
                if (cart[index].quantity <= 0) {
                    cart.splice(index, 1);
                }
                
                localStorage.setItem('sol_cart', JSON.stringify(cart));
                updateCartCount();
                renderCart();
            }
        };

        // Remove item
        window.removeItem = function(index) {
            const item = cart[index];
            cart.splice(index, 1);
            localStorage.setItem('sol_cart', JSON.stringify(cart));
            updateCartCount();
            renderCart();
            showNotification(`🗑️ ${item.name} removed from cart`);
        };

        // Update cart count
        function updateCartCount() {
            const count = cart.reduce((sum, item) => sum + item.quantity, 0);
            document.getElementById('cartCount').textContent = count;
        }

        // Toggle cart
        window.toggleCart = function() {
            document.getElementById('cartSidebar').classList.toggle('open');
            document.getElementById('overlay').classList.toggle('show');
        };

        // Show notification
        function showNotification(message) {
            const notif = document.getElementById('notification');
            document.getElementById('notificationMessage').textContent = message;
            notif.classList.add('show');
            setTimeout(() => {
                notif.classList.remove('show');
            }, 3000);
        }

        // Open checkout modal
        window.openCheckout = function() {
            if (cart.length === 0) {
                showNotification('Your cart is empty!');
                return;
            }
            
            document.getElementById('checkoutModal').classList.add('show');
            document.getElementById('overlay').classList.add('show');
        };

        // Close checkout modal
        window.closeCheckout = function() {
            document.getElementById('checkoutModal').classList.remove('show');
            if (!document.getElementById('cartSidebar').classList.contains('open')) {
                document.getElementById('overlay').classList.remove('show');
            }
        };

        // Select payment method
        window.selectPayment = function(method) {
            selectedPayment = method;
            
            // Update UI
            document.querySelectorAll('.payment-method').forEach(el => {
                el.classList.remove('selected');
            });
            document.getElementById(`payment${method.toUpperCase()}`).classList.add('selected');
        };

        // Place order
        window.placeOrder = function() {
            // Get form data
            const name = document.getElementById('checkoutName').value;
            const email = document.getElementById('checkoutEmail').value;
            const phone = document.getElementById('checkoutPhone').value;
            const address = document.getElementById('checkoutAddress').value;

            // Validate
            if (!name || !email || !phone || !address) {
                showNotification('Please fill all fields!');
                return;
            }

            // Calculate total
            const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);

            // Create order
            const order = {
                id: 'ORD' + Date.now(),
                customer: { name, email, phone, address },
                items: [...cart],
                total: total,
                payment: selectedPayment,
                date: new Date().toISOString(),
                status: 'confirmed'
            };

            // Save order to localStorage
            const orders = JSON.parse(localStorage.getItem('sol_orders')) || [];
            orders.push(order);
            localStorage.setItem('sol_orders', JSON.stringify(orders));

            // Clear cart
            cart = [];
            localStorage.setItem('sol_cart', JSON.stringify(cart));
            updateCartCount();
            renderCart();

            // Close modals
            closeCheckout();
            toggleCart();

            // Show success message
            showNotification(`✅ Order placed! Total: ₹${total}`);
            
            // Clear form
            document.getElementById('checkoutName').value = '';
            document.getElementById('checkoutEmail').value = '';
            document.getElementById('checkoutPhone').value = '';
            document.getElementById('checkoutAddress').value = '';
        };

        // Check for updates from admin panel
        function checkForUpdates() {
            const savedProducts = localStorage.getItem('sol_products');
            if (savedProducts) {
                const newProducts = JSON.parse(savedProducts);
                if (JSON.stringify(products) !== JSON.stringify(newProducts)) {
                    products = newProducts;
                    renderProducts();
                    showNotification('✨ New products available!', 'info');
                }
            }
        }

        // Initialize
        loadProducts();
        updateCartCount();
        renderCart();

        // Check for updates every 2 seconds
        setInterval(checkForUpdates, 2000);

        // Listen for storage events (updates from admin panel)
        window.addEventListener('storage', function(e) {
            if (e.key === 'sol_products') {
                products = JSON.parse(e.newValue);
                renderProducts();
                showNotification('✨ Products updated!', 'info');
            }
        });
