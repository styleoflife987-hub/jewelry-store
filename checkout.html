// Add this to your checkout.html processOrder function async function processOrder() { // ... validation code ... const orderData = { name: name, phone: phone, email: email || '', address: fullAddress, items: cart, subtotal: subtotal, tax: tax, shipping: shipping, total: total, status: 'Pending', date: new Date().toISOString() }; try { // Save to Excel via live sync const result = await placeOrderInExcel(orderData); if (result.success) { // Save locally as backup const orders = JSON.parse(localStorage.getItem('orders') || '[]'); orders.push({...orderData, orderId: result.orderId}); localStorage.setItem('orders', JSON.stringify(orders)); // Clear cart localStorage.removeItem('cart'); updateCartCount(); alert(`✅ Order placed successfully! Order ID: ${result.orderId}`); window.location.href = `track.html?order=${result.orderId}`; } else { throw new Error(result.error); } } catch (error) { // Fallback to local storage const fallbackOrderId = 'ORD' + Date.now(); const orders = JSON.parse(localStorage.getItem('orders') || '[]'); orders.push({...orderData, orderId: fallbackOrderId}); localStorage.setItem('orders', JSON.stringify(orders)); localStorage.removeItem('cart'); updateCartCount(); alert(`⚠️ Order saved locally. Order ID: ${fallbackOrderId}`); window.location.href = `track.html?order=${fallbackOrderId}`; } }
